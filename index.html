Object data type

5 primitive data types : number, string, boolean, undefined and null.
An object is an unordered list of primitive data types stored as name-value pairs within properties or methods(functions)

Primitive data type

var name = "Fred";
var anotherName = name;
name = "Max"
log(anotherName) //Fred
log(name) //Max

Reference data type

var person = {name:"Fred"};
var anotherPerson = person;
person.name = "Max";
log(anotherPerson.name) //Max
log(person.name) //Max

var userChoice = prompt("Do you choose rock, paper or scissors?");
var computerChoice = Math.random();
if (computerChoice < 0.34) {
computerChoice = "rock";
} else if(computerChoice <= 0.67) {
computerChoice = "paper";
} else {
computerChoice = "scissors";
}

console.log("Computer: " + computerChoice);

function compare(choice1, choice2) {
if (choice1 === choice2) {
console.log("The result is a tie!");
} else if (choice1 === "rock") {
if (choice2 === "scissors"){
console.log("rock wins");
} else {
console.log("paper wins");
}
}
}
